});

/***********************************************************
 * 3) LocalStorageの読み書き
 ***********************************************************/
function saveMemorizedStatus() {
  const memorizedMap = {};
  words.forEach(w => {
    memorizedMap[w.id] = w.memorized;
  });
  localStorage.setItem("memorizedMap", JSON.stringify(memorizedMap));
}

function loadMemorizedStatus() {
  const data = localStorage.getItem("memorizedMap");
  if (!data) return;
  const memorizedMap = JSON.parse(data);
  words.forEach(w => {
    if (memorizedMap[w.id] !== undefined) {
      w.memorized = memorizedMap[w.id];
    }
  });
}

/***********************************************************
 * 4) 単語検索
 ***********************************************************/
const searchInput = document.getElementById("searchInput");
const searchBtn = document.getElementById("searchBtn");
const searchResults = document.getElementById("searchResults");

searchBtn.addEventListener("click", () => {
  const query = searchInput.value.trim();
  if (!query) {
    searchResults.innerHTML = "<p>検索キーワードを入力してください。</p>";
    return;
  }
  const filtered = words.filter(item =>
    item.chinese.includes(query) || item.pinyin.includes(query)
  );
  renderSearchResults(filtered);
});

function renderSearchResults(list) {
  if (list.length === 0) {
    searchResults.innerHTML = "<p>該当する単語が見つかりません。</p>";
    return;
  }
  let html = `
    <table>
      <thead>
        <tr>
          <th>中国語</th>
          <th>ピンイン</th>
          <th>品詞</th>
          <th>和訳</th>
          <th>暗記済み?</th>
        </tr>
      </thead>
      <tbody>
  `;
  list.forEach(item => {
    html += `
      <tr>
        <td>${item.chinese}</td>
        <td>${item.pinyin}</td>
        <td>${item.partOfSpeech}</td>
        <td>${item.meaning}</td>
        <td>
          <input type="checkbox" class="memorized-checkbox" data-id="${item.id}"
            ${item.memorized ? "checked" : ""}>
        </td>
      </tr>
    `;
  });
  html += "</tbody></table>";
  searchResults.innerHTML = html;

  // チェックボックス イベント付与
  const checkboxes = searchResults.querySelectorAll(".memorized-checkbox");
  checkboxes.forEach(chk => {
    chk.addEventListener("change", onCheckboxChange);
  });
}

/***********************************************************
 * 5) 全単語一覧 表示
 ***********************************************************/
const allWordsList = document.getElementById("allWordsList");

function displayAllWords() {
  let html = `
    <table>
      <thead>
        <tr>
          <th>中国語</th>
          <th>ピンイン</th>
          <th>品詞</th>
          <th>和訳</th>
          <th>暗記済み?</th>
        </tr>
      </thead>
      <tbody>
  `;
  words.forEach(item => {
    html += `
      <tr>
        <td>${item.chinese}</td>
        <td>${item.pinyin}</td>
        <td>${item.partOfSpeech}</td>
        <td>${item.meaning}</td>
        <td>
          <input type="checkbox" class="memorized-checkbox" data-id="${item.id}"
            ${item.memorized ? "checked" : ""}>
        </td>
      </tr>
    `;
  });
  html += "</tbody></table>";
  allWordsList.innerHTML = html;

  const checkboxes = allWordsList.querySelectorAll(".memorized-checkbox");
  checkboxes.forEach(chk => {
    chk.addEventListener("change", onCheckboxChange);
  });
}

/***********************************************************
 * チェックボックスの変更イベント
 ***********************************************************/
function onCheckboxChange(e) {
  const checkbox = e.target;
  const wordId = parseInt(checkbox.dataset.id, 10);
  const targetWord = words.find(w => w.id === wordId);
  if (targetWord) {
    targetWord.memorized = checkbox.checked;
    saveMemorizedStatus();
  }
}

/***********************************************************
 * 6) フラッシュカード
 *   - 通常モード：表=中国語, 裏=ピンイン/品詞/和訳
 *   - 逆モード：表=日本語, 裏=中国語/ピンイン/品詞
 *   - 例文はカードが裏面になったときのみ表示
 ***********************************************************/
let currentIndex = -1;
let currentWordList = [];
let isReverseMode = false; // false=通常(中→日)、true=逆モード(日→中)

const flashcard = document.getElementById("flashcard");
const flashcardFront = document.getElementById("flashcardFront");
const flashcardBackPinyin = document.getElementById("pinyin");
const flashcardBackPos = document.getElementById("partOfSpeech");
const flashcardBackMeaning = document.getElementById("meaning");

const exampleBox = document.getElementById("exampleBox");
const exampleContent = document.getElementById("exampleContent");

const flipBtn = document.getElementById("flipBtn");
const nextBtn = document.getElementById("nextBtn");
const startRandomAllBtn = document.getElementById("startRandomAllBtn");
const startRandomMemorizedBtn = document.getElementById("startRandomMemorizedBtn");
const startRandomUnmemorizedBtn = document.getElementById("startRandomUnmemorizedBtn");
const reverseModeBtn = document.getElementById("reverseModeBtn");

// ==== カードをめくる ====
flipBtn.addEventListener("click", () => {
  flashcard.classList.toggle("show-back");
  // 裏面になったら例文表示、表面に戻したら非表示
  if (flashcard.classList.contains("show-back")) {
    exampleBox.style.display = "block";
  } else {
    exampleBox.style.display = "none";
  }
});

// ==== 次へ ====
nextBtn.addEventListener("click", () => {
  showRandomCard();
});

// ==== ランダム出題ボタン ====
startRandomAllBtn.addEventListener("click", () => {
  currentWordList = words;
  showRandomCard();
});
startRandomMemorizedBtn.addEventListener("click", () => {
  const memorizedOnly = words.filter(w => w.memorized);
  if (memorizedOnly.length === 0) {
    alert("暗記済みの単語がありません。");
    return;
  }
  currentWordList = memorizedOnly;
  showRandomCard();
});
startRandomUnmemorizedBtn.addEventListener("click", () => {
  const unmemorizedOnly = words.filter(w => !w.memorized);
  if (unmemorizedOnly.length === 0) {
    alert("未暗記の単語がありません。");
    return;
  }
  currentWordList = unmemorizedOnly;
  showRandomCard();
});

// ==== モード切り替えボタン ====
reverseModeBtn.addEventListener("click", () => {
  isReverseMode = !isReverseMode;
  if (isReverseMode) {
    reverseModeBtn.textContent = "中日モードに切り替え";
  } else {
    reverseModeBtn.textContent = "日中モードに切り替え";
  }
  // いま出題中の単語も反映し直す
  if (currentIndex >= 0) {
    const wordData = words[currentIndex];
    renderFlashcard(wordData);
  }
});

/**
 * ランダムに1単語ピック
 */
function showRandomCard() {
  if (!currentWordList || currentWordList.length === 0) {
    alert("出題できる単語がありません。");
    return;
  }
  const randomIndex = Math.floor(Math.random() * currentWordList.length);
  const wordData = currentWordList[randomIndex];
  currentIndex = words.findIndex(w => w.id === wordData.id);

  renderFlashcard(wordData);
}

/**
 * カードの表示更新
 */
function renderFlashcard(wordData) {
  // まず表面にリセット
  flashcard.classList.remove("show-back");
  exampleBox.style.display = "none"; // 例文も隠す

  if (!isReverseMode) {
    // ===== 通常モード: 表=中国語, 裏=ピンイン/品詞/意味 =====
    flashcardFront.textContent = wordData.chinese;
    flashcardBackPinyin.textContent = `ピンイン: ${wordData.pinyin}`;
    flashcardBackPos.textContent = `品詞: ${wordData.partOfSpeech}`;
    flashcardBackMeaning.textContent = `意味: ${wordData.meaning}`;
  } else {
    // ===== 逆モード: 表=日本語, 裏=中国語/ピンイン/品詞 =====
    flashcardFront.textContent = wordData.meaning; // 表面に日本語
    flashcardBackPinyin.textContent = `ピンイン: ${wordData.pinyin}`;
    flashcardBackPos.textContent = `品詞: ${wordData.partOfSpeech}`;
    flashcardBackMeaning.textContent = `中国語: ${wordData.chinese}`;
  }

  // 例文は裏面になったときだけ表示するので、例文の中身だけ更新
  exampleContent.innerHTML = `
    <p><strong>${wordData.example.sentence}</strong></p>
    <p>${wordData.example.pinyin}</p>
    <p>(${wordData.example.translation})</p>
  `;
}
